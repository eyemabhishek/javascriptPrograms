
// Program to check if number betwwen 20 - 30 is even or odd and 
// display in below fashion
// "20 is even"
// "21 is odd"
// "22 is even"


for (let i = 20; i <= 30; i++) {
    if (i % 2 == 0) {
        console.log(i, " is even");
    }
    else {
        console.log(i, " is odd");
    }

}
